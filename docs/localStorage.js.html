/**
 * @module localStorageUtils
 * @description Utilities for handling LocalStorage actions like saving, retrieving and removing tokens and users.
 */


/**
 * Saves a value to localStorage. Removes the item if the value is null or undefined.
 * 
 * @function
 * @param {string} key - The localStorage key.
 * @param {*} value - The value to be saved.
 * @example
 * saveToLocalStorage('user', { name: 'John' });
 */
function saveToLocalStorage(key, value) {
  if (value !== null && value !== undefined) {
    localStorage.setItem(key, JSON.stringify(value));

  } else {
    localStorage.removeItem(key);
  }
}

/**
 * Gets a value from localStorage. Returns a default if not found or JSON is invalid.
 * 
 * @function
 * @param {string} key - The localStorage key.
 * @param {*} [defaultValue=null] - Default value if key not found or parsing fails.
 * @returns {*} The parsed value or defaultValue.
 * @example
 * const user = getFromLocalStorage('user', {});
 */
function getFromLocalStorage(key, defaultValue = null) {
  const result = localStorage.getItem(key);
  if (result) {
    try {
      return JSON.parse(result); //intentamos parsear el resultado
    } catch (error) {
      console.error("Error parsing JSON from localStorage:", error);
      localStorage.removeItem(key); // eliminamos el item si no se puede parsear  
      return defaultValue;
    }
  } else {
    return defaultValue;
  }
}

/**
 * Removes a value from localStorage.
 * @function
 * @param {string} key - The localStorage key to remove.
 * @example
 * removeFromLocalStorage('token');
 */
function removeFromLocalStorage(key) {
  localStorage.removeItem(key);
}

/**
 * @module tokenUtils
 * @description Utilities for handling the user token.
 */

/**
 * Saves a token, calling to the saveToLocalStorage function
 * @function
 * @param {string} token - The auth token.
 */
function saveToken(token) {
  if (token) {
    saveToLocalStorage("token", token);
  }
}

/**
 * Get a token, calling to the getFromLocalStorage function
 * @function
 * @returns {string|null} The token or null if not found.
 */

function getToken() {
  return getFromLocalStorage("token", null);
}

/**
 * Remove a token, calling to the removeFromLocalStorage function
 * @function
 */
function removeToken() {
  removeFromLocalStorage("token");
}

/**
 * Saves a user to local storage
 * @function
 * @param {*} user - The user object
 */
function saveUser(user) {
  saveToLocalStorage("user", user);
}

/**
 * Get a user from local storage
 * @function
 * @returns {Object|null} The user or null if not found
 */
function getUser() {
  return getFromLocalStorage("user", null);
}

/**
 * Remove a user from local storage
 * @function
 */
function removeUser() {
  removeFromLocalStorage("user");
}

export { saveToLocalStorage, getFromLocalStorage, saveToken, getToken, removeToken, saveUser, getUser, removeUser };
