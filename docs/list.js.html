import FetchData from "./fetch";

/**
 * @module listUtils
 * @description Utilities for handling lists with CRUD operations calling the FetchData function.
 */



/**
 * Get all lists. It calls the FetchData function with the route /list
 * @function
 * @returns -> lists
 */

const getAllLists = async () => {
const lists = await FetchData("/list");
  return lists;
};

/**
 * Get a list by id. It calls the FetchData function with the route /list/:id
 * @function
 * @param {string} listId - The id of the list
 * @returns -> list
 */

const getListById = async (listId) => {
  const list = await FetchData(`/list/${listId}`);
  return list;
};

/**
 * Get a list by project id. It calls the FetchData function with the route /list/project/:projectId
 * @function
 * @param {string} projectId - The id of the project
 * @returns -> list
 */

const getListByProjectId = async (projectId) => {
  const lists = await FetchData(`/list/project/${projectId}`);
  return lists;
};

/**
 * Create a list. It calls the FetchData function with the route /list with the method POST
 * @function
 * @param {*} list - The list to create
 * @returns -> new list
 */
const createList = async (list) => {
  const newList = await FetchData("/list", "POST", list);
  return newList;
};

/**
 * Remove a list. It calls the FetchData function with the route /list/:id with the method DELETE
 * @function
 * @param {string} listId - The id of the list
 * @returns -> list
 */
const removeList = async (listId) => {
  const response = await FetchData(`/list/${listId}`, "DELETE");
  return response;
}

/**
 * Update a list. It calls the FetchData function with the route /list/:id with the method PUT
 * @function
 * @param {string} listId  - The id of the list
 * @param {*} list - The list to update
 * @returns -> updated list
 */
const updateList = async (listId, list) => {
  const updatedList = await FetchData(`/list/${listId}`, "PUT", list);
  return updatedList;
};

export { getAllLists, getListById, getListByProjectId, createList, removeList, updateList };
